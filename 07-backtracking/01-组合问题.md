    def combine(self, n: int, k: int) -> List[List[int]]:
        res = []

        def backtracking(n, k, l: list):
            if k == 0:
                res.append(l)
                return

            for i in range(n, k-1, -1): # å‰ªæä¼˜åŒ–, range: [k, k+1, ..., n]

        backtracking(n, k, [])
        return res
