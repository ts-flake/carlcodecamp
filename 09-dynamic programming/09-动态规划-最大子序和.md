class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        dp = [0]*len(nums)
        dp[0] = _max = nums[0]
        for i in range(1,len(nums)):
            dp[i] = max(nums[i], dp[i-1]+nums[i])
            _max = max(_max, dp[i]) # éœ€è¦æ›´æ–°æœ€å¤§ç»“æœ
        return _max
