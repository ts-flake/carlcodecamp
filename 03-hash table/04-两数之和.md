```python
class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        num2ind = dict()
        for i,n in enumerate(nums):
            if n not in num2ind:
                num2ind[n] = [i]
            else:
                num2ind[n].append(i)
        
        for n1 in nums:
            n2 = target - n1
            ind1 = num2ind[n1].pop()
            ind2 = num2ind.get(n2, [])
            if ind2:
                return ind1, ind2[0]
```
