class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        if len(prices) == 0:
            return 0
        dp = [[0] * 5 for _ in range(len(prices))]
        dp[0][1] = -prices[0]
        dp[0][3] = -prices[0]
        for i in range(1, len(prices)):
            dp[i][0] = dp[i-1][0]
            dp[i][1] = max(dp[i-1][1], dp[i-1][0] - prices[i]) # Á¨¨‰∏ÄÊ¨°‰π∞ÂÖ•
            dp[i][2] = max(dp[i-1][2], dp[i-1][1] + prices[i]) # Á¨¨‰∏ÄÊ¨°ÂçñÂá∫
            dp[i][3] = max(dp[i-1][3], dp[i-1][2] - prices[i]) # Á¨¨‰∫åÊ¨°‰π∞ÂÖ•
            dp[i][4] = max(dp[i-1][4], dp[i-1][3] + prices[i]) # Á¨¨‰∫åÊ¨°ÂçñÂá∫
        return dp[-1][4]
