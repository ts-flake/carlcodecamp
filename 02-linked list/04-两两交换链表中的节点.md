#         self.next = next
class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        if not head:
            return None
        if not head.next:
            return head
        # ä¸¤ä¸ªèŠ‚ç‚¹åŠä»¥ä¸Š

        cur = prev.next.next
        head = prev.next
        while prev.next:
            prev.next.next = prev
            # null æˆ–æœ«èŠ‚ç‚¹
            if not cur or not cur.next:
                prev.next = cur
                return head
            else:
                prev.next = cur.next
            prev = cur
            cur = cur.next.next
        return head
