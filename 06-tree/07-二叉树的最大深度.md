class Solution:
    def maxDepth(self, root: Optional[TreeNode]) -> int:
        def _depth(node, lvl):
            if not node: return lvl
            return 1 + max(_depth(node.left, lvl), _depth(node.right, lvl))
        return _depth(root, 0)
