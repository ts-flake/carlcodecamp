## 题目
[707.设计链表 | 力控题目](https://leetcode.cn/problems/design-linked-list/description/)

代码实现单/双链表, 并包含查询, 增加和删除操作. 假设链表下标从 0 开始.

## 初步解答
下面实现了单链表. 主要还是考察对链表的定义及操作.

```python
class Node:
    def __init__(self, val=0, next=None):
        self.val=val
        self.next=next

class MyLinkedList:
    def __init__(self):
        self.dummy_head = Node()
        self._len = 0

    def get(self, index: int) -> int:
        if index > self._len - 1 or index < 0:
            return -1
        
        cur = self.dummy_head.next
        for _ in range(index):
            cur = cur.next
        return cur.val
        
    def addAtHead(self, val: int) -> None:
        head = Node(val)
        head.next = self.dummy_head.next
        self.dummy_head.next = head
        self._len += 1
        
    def addAtTail(self, val: int) -> None:
        if not hasattr(self, '_tail'):
            cur = self.dummy_head
            while cur.next:
                cur = cur.next
            cur.next = Node(val)
            self._tail = cur.next
        else:
            self._tail.next = Node(val)
            self._tail = self._tail.next
        self._len += 1

    def addAtIndex(self, index: int, val: int) -> None:
        if index > self._len or index < 0:
            return
        if index == self._len:
            return self.addAtTail(val)
        if index == 0:
            return self.addAtHead(val)
        
        cur = self.dummy_head.next
        for _ in range(index-1):
            cur = cur.next
        new = Node(val)
        new.next = cur.next
        cur.next = new
        self._len += 1

    def deleteAtIndex(self, index: int) -> None:
        if index > self._len - 1 or index < 0:
            return
        cur = self.dummy_head
        for _ in range(index):
            cur = cur.next
        cur.next = cur.next.next
        
        if index == self._len - 1:
            self._tail = cur
        self._len -= 1


# Your MyLinkedList object will be instantiated and called as such:
# obj = MyLinkedList()
# param_1 = obj.get(index)
# obj.addAtHead(val)
# obj.addAtTail(val)
# obj.addAtIndex(index,val)
# obj.deleteAtIndex(index)
```

## 参考解答
自己写的和参考代码基本一致, 但对 `addTail` 做了点改进, 让 `MyLinkedList` 类记录当前的末尾节点, 下次再操作时时间效率上从 O(n) 提升到 O(1).
```python
class ListNode:
    def __init__(self, val=0, next=None):
        self.val = val
        self.next = next
        
class MyLinkedList:
    def __init__(self):
        self.dummy_head = ListNode()
        self.size = 0

    def get(self, index: int) -> int:
        if index < 0 or index >= self.size:
            return -1
        
        current = self.dummy_head.next
        for i in range(index):
            current = current.next
            
        return current.val

    def addAtHead(self, val: int) -> None:
        self.dummy_head.next = ListNode(val, self.dummy_head.next)
        self.size += 1

    def addAtTail(self, val: int) -> None:
        current = self.dummy_head
        while current.next:
            current = current.next
        current.next = ListNode(val)
        self.size += 1

    def addAtIndex(self, index: int, val: int) -> None:
        if index < 0 or index > self.size:
            return
        
        current = self.dummy_head
        for i in range(index):
            current = current.next
        current.next = ListNode(val, current.next)
        self.size += 1

    def deleteAtIndex(self, index: int) -> None:
        if index < 0 or index >= self.size:
            return
        
        current = self.dummy_head
        for i in range(index):
            current = current.next
        current.next = current.next.next
        self.size -= 1

# obj = MyLinkedList()
# param_1 = obj.get(index)
# obj.addAtHead(val)
# obj.addAtTail(val)
# obj.addAtIndex(index,val)
# obj.deleteAtIndex(index)
```

## 心得
再次用到 dummy head 的技巧, 本题用于提升对链表的熟悉度. ⚠️代码上最要注意的是 while 循环, 明确起始节点在哪, 循环结束后的节点在哪. 不然在后续操作时, e.g. `cur.val/next` 容易出现 NoneType 没有属性的错误.
