class Solution:
    def reverseStr(self, s: str, k: int) -> str:
        s = list(s)
        left,right = 0,2*k-1
        while right < len(s)-1:
            s[left:left+k] = reversed(s[left:left+k])
            left += 2*k
            right += 2*k
        # å¤„ç†å‰©ä½™çš„å­—ç¬¦

        return ''.join(s)
